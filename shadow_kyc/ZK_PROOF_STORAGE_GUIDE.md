# ZK Proof Storage - Integration Guide

## Overview

The `ZKProofStorage` contract stores ZK proof hashes on Sepolia testnet, allowing users to retrieve them for verification purposes. After a ZK proof is generated by the backend, it's hashed and stored on-chain for permanent records.

## Smart Contract Details

### ZKProofStorage.sol - Key Features

1. **Proof Hash Storage**: Stores keccak256 hash of each proof
2. **User Mapping**: Maps user addresses to their latest proof hash
3. **Proof History**: Maintains complete history of all proofs with `ProofRecord`
4. **On-Chain Verification**: Integrates with `Groth16Verifier` for proof validation
5. **KYC Level Tracking**: Automatically calculates KYC level from proof flags
6. **Event Logging**: Emits events for all proof operations

### Data Structures

```solidity
struct ProofRecord {
    bytes32 proofHash;           // Keccak256 hash of the proof
    uint256 timestamp;           // When proof was submitted
    bool verified;               // On-chain verification status
    uint8 kycLevel;             // Calculated KYC level (0-4)
    bytes32 credentialHash;     // Backend credential hash
    uint256[3] publicSignals;   // ZK public signals
}

struct ProofDetails {
    address user;               // User address
    bytes32 credentialHash;     // Backend credential hash
    uint256[3] publicSignals;   // Public signals
    uint256 timestamp;          // Submission time
    bool isVerified;           // Verification status
    uint8 kycLevel;            // KYC level
    string proofMetadata;      // JSON metadata
}
```

## Backend Integration

### Update Backend Routes

#### File: `backend/src/routes/kyc.ts`

```typescript
import express, { Request, Response } from "express";
import { hashCredential } from "../utils/poseidon";
import { signCredentialHash } from "../utils/issuer";
import { generateProof } from "../utils/prover";
import { ethers } from "ethers";

const router = express.Router();

// Initialize ethers provider and contract
const provider = new ethers.JsonRpcProvider(
  process.env.SEPOLIA_RPC_URL || "https://sepolia.infura.io/v3/..."
);

const wallet = new ethers.Wallet(
  process.env.WALLET_PRIVATE_KEY || "",
  provider
);

const ZK_PROOF_STORAGE_ABI = require("../abis/ZKProofStorage.json");
const zkProofContract = new ethers.Contract(
  process.env.ZKPROOF_STORAGE_ADDRESS || "",
  ZK_PROOF_STORAGE_ABI,
  wallet
);

interface KYCRequest {
  ageNatOK: number;
  govtIdOK: number;
  faceOK: number;
  livenessOK: number;
}

interface ProofResponse {
  proof: {
    pi_a: string[];
    pi_b: string[][];
    pi_c: string[];
    protocol: string;
    curve: string;
  };
  publicSignals: string[];
  credentialHash: string;
  proofHash?: string;
  transactionHash?: string;
  chainTransactionHash?: string;
}

/**
 * POST /kyc/submit
 * Submit KYC flags and generate ZK proof
 * Also stores proof hash on Sepolia testnet
 */
router.post("/submit", async (req: Request, res: Response) => {
  try {
    const { ageNatOK, govtIdOK, faceOK, livenessOK, storeOnChain } = req.body;
    const salt = Math.floor(Math.random() * 1e6);

    const flags = {
      ageNatOK: Number(ageNatOK),
      govtIdOK: Number(govtIdOK),
      faceOK: Number(faceOK),
      livenessOK: Number(livenessOK),
    };

    // Generate credential hash and proof
    const hash = hashCredential(flags, salt);
    const sig = signCredentialHash(hash, process.env.ISSUER_PRIVATE_KEY || "");

    const input = {
      ageNatOK: ageNatOK.toString(),
      govtIdOK: govtIdOK.toString(),
      faceOK: faceOK.toString(),
      livenessOK: livenessOK.toString(),
      salt: salt.toString(),
      sigR8x: sig.sigR8x.toString(),
      sigR8y: sig.sigR8y.toString(),
      sigS: sig.sigS.toString(),
      issuerPubKey: sig.pubKey.map((x: any) => x.toString()),
    };

    console.log("Generating ZK proof...");
    const proofData = await generateProof(input);

    const response: ProofResponse = {
      proof: proofData.proof,
      publicSignals: proofData.publicSignals,
      credentialHash: hash,
    };

    // Store proof hash on-chain if requested
    if (storeOnChain) {
      console.log("Storing proof hash on Sepolia testnet...");
      
      try {
        // Prepare proof submission data
        const proofMetadata = JSON.stringify({
          ageNatOK,
          govtIdOK,
          faceOK,
          livenessOK,
          salt,
          timestamp: new Date().toISOString(),
        });

        // Submit proof to blockchain
        const tx = await zkProofContract.submitProof(
          proofData.proof.pi_a.slice(0, 2), // uint[2]
          [
            proofData.proof.pi_b[0].slice(0, 2),
            proofData.proof.pi_b[1].slice(0, 2),
          ], // uint[2][2]
          proofData.proof.pi_c.slice(0, 2), // uint[2]
          proofData.publicSignals.slice(0, 3).map(BigInt), // uint256[3]
          hash, // bytes32
          proofMetadata // string
        );

        console.log("Transaction hash:", tx.hash);
        response.transactionHash = tx.hash;
        response.chainTransactionHash = tx.hash;

        // Wait for transaction confirmation
        const receipt = await tx.wait();
        console.log("Proof stored on-chain. Block:", receipt?.blockNumber);
      } catch (chainError) {
        console.error("Error storing proof on-chain:", chainError);
        // Don't fail the request, just log the error
        // Proof is still generated locally
      }
    }

    res.json(response);
  } catch (err) {
    console.error("KYC submit error:", err);
    res
      .status(500)
      .json({
        error: err instanceof Error ? err.message : String(err),
      });
  }
});

/**
 * GET /kyc/proof-hash/:userAddress
 * Retrieve proof hash for a user from blockchain
 */
router.get("/proof-hash/:userAddress", async (req: Request, res: Response) => {
  try {
    const { userAddress } = req.params;

    // Validate address
    if (!ethers.isAddress(userAddress)) {
      return res.status(400).json({ error: "Invalid Ethereum address" });
    }

    // Get proof hash from contract
    const proofHash = await zkProofContract.getProofHash(userAddress);

    if (proofHash === ethers.ZeroHash) {
      return res
        .status(404)
        .json({ error: "No proof found for this address" });
    }

    res.json({
      userAddress,
      proofHash,
      chainId: 11155111, // Sepolia
      contractAddress: process.env.ZKPROOF_STORAGE_ADDRESS,
    });
  } catch (err) {
    console.error("Error retrieving proof hash:", err);
    res
      .status(500)
      .json({
        error: err instanceof Error ? err.message : String(err),
      });
  }
});

/**
 * GET /kyc/proof-details/:proofHash
 * Get detailed proof information from blockchain
 */
router.get("/proof-details/:proofHash", async (req: Request, res: Response) => {
  try {
    const { proofHash } = req.params;

    // Validate hash format
    if (!/^0x[0-9a-fA-F]{64}$/.test(proofHash)) {
      return res.status(400).json({ error: "Invalid proof hash format" });
    }

    // Get proof details
    const proofDetails = await zkProofContract.getProofDetails(proofHash);

    res.json({
      user: proofDetails.user,
      credentialHash: proofDetails.credentialHash,
      publicSignals: proofDetails.publicSignals,
      timestamp: Number(proofDetails.timestamp),
      isVerified: proofDetails.isVerified,
      kycLevel: proofDetails.kycLevel,
      proofMetadata: JSON.parse(proofDetails.proofMetadata),
    });
  } catch (err) {
    console.error("Error retrieving proof details:", err);
    res
      .status(500)
      .json({
        error: err instanceof Error ? err.message : String(err),
      });
  }
});

/**
 * POST /kyc/verify-onchain
 * Verify a proof on-chain using Groth16Verifier
 */
router.post("/verify-onchain", async (req: Request, res: Response) => {
  try {
    const { proofHash, proof } = req.body;

    if (!proofHash || !proof) {
      return res
        .status(400)
        .json({ error: "Missing proofHash or proof data" });
    }

    // Verify on chain
    const isValid = await zkProofContract.verifyProof(
      proofHash,
      proof.pi_a.slice(0, 2),
      [proof.pi_b[0].slice(0, 2), proof.pi_b[1].slice(0, 2)],
      proof.pi_c.slice(0, 2)
    );

    res.json({
      proofHash,
      isValid,
      verifiedAt: new Date().toISOString(),
      chainId: 11155111,
    });
  } catch (err) {
    console.error("Error verifying proof on-chain:", err);
    res
      .status(500)
      .json({
        error: err instanceof Error ? err.message : String(err),
      });
  }
});

/**
 * GET /kyc/user-kyc-level/:userAddress
 * Get user's calculated KYC level from blockchain
 */
router.get(
  "/user-kyc-level/:userAddress",
  async (req: Request, res: Response) => {
    try {
      const { userAddress } = req.params;

      if (!ethers.isAddress(userAddress)) {
        return res.status(400).json({ error: "Invalid Ethereum address" });
      }

      const kycLevel = await zkProofContract.getUserKYCLevel(userAddress);

      const levelNames = [
        "No Verification",
        "Age/Nationality",
        "Government ID",
        "Face Verified",
        "Liveness Verified (Complete)",
      ];

      res.json({
        userAddress,
        kycLevel,
        levelName: levelNames[kycLevel] || "Unknown",
      });
    } catch (err) {
      console.error("Error getting KYC level:", err);
      res
        .status(500)
        .json({
          error: err instanceof Error ? err.message : String(err),
        });
    }
  }
);

export default router;
```

```typescript
import axios from "axios";

const API_BASE = process.env.REACT_APP_API_URL || "http://localhost:3000";

export interface KYCFlags {
  ageNatOK: number;
  govtIdOK: number;
  faceOK: number;
  livenessOK: number;
}

export interface ProofData {
  proof: {
    pi_a: string[];
    pi_b: string[][];
    pi_c: string[];
    protocol: string;
    curve: string;
  };
  publicSignals: string[];
  credentialHash: string;
  proofHash?: string;
  chainTransactionHash?: string;

export async function submitKYC(
  flags: KYCFlags,
  storeOnChain = true
): Promise<ProofData> {
  const response = await axios.post(`${API_BASE}/kyc/submit`, {
    ...flags,
    storeOnChain,
  });
  return response.data;
}

export async function retrieveProofHash(userAddress: string) {
  const response = await axios.get(`${API_BASE}/kyc/proof-hash/${userAddress}`);
  return response.data;
}

export async function getProofDetails(proofHash: string) {
  const response = await axios.get(
    `${API_BASE}/kyc/proof-details/${proofHash}`
  );
  return response.data;
}

export async function verifyProofOnChain(
  proofHash: string,
  proof: any
) {
  const response = await axios.post(`${API_BASE}/kyc/verify-onchain`, {
    proofHash,
    proof,
  });
  return response.data;
}

export async function getUserKYCLevel(userAddress: string) {
  const response = await axios.get(
    `${API_BASE}/kyc/user-kyc-level/${userAddress}`
  );
  return response.data;
}
```

## Troubleshooting

### Transaction Failed
- Check account has enough Sepolia ETH
- Verify contract addresses are correct
- Check RPC endpoint is working

### Proof Not Stored
- Ensure `storeOnChain` parameter is `true`
- Check wallet has gas
- Verify contract deployment

### Can't Retrieve Proof
- Verify you're using correct Ethereum address
- Check proof hash format (0x prefix + 64 hex chars)
- Ensure proof was submitted before retrieval
